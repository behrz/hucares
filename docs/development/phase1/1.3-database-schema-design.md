// src/models/User.ts
import { Schema, model } from 'mongoose';

const UserSchema = new Schema({
  email: { type: String, required: true, unique: true },
  password: { type: String, required: true },
  firstName: { type: String, required: true },
  lastName: { type: String, required: true },
  role: { type: String, enum: ['user', 'admin'], default: 'user' },
  lastLogin: { type: Date },
  isActive: { type: Boolean, default: true },
  profile: {
    avatar: String,
    timezone: { type: String, default: 'UTC' },
    notifications: { type: Boolean, default: true },
  },
}, { timestamps: true });

// Indexes for performance
UserSchema.index({ email: 1 });
UserSchema.index({ createdAt: 1 });

export const User = model('User', UserSchema);
```

const CheckInSchema = new Schema({
  userId: { type: Schema.Types.ObjectId, ref: 'User', required: true },
  date: { type: Date, required: true },
  mood: { 
    type: Number, 
    required: true,
    min: 1,
    max: 10 
  },
  energy: { 
    type: Number, 
    required: true,
    min: 1,
    max: 10 
  },
  stress: { 
    type: Number, 
    required: true,
    min: 1,
    max: 10 
  },
  notes: { type: String, maxlength: 500 },
  tags: [{ type: String, trim: true }],
  isPrivate: { type: Boolean, default: false },
}, { timestamps: true });

// Compound index for unique daily check-ins
CheckInSchema.index({ userId: 1, date: 1 }, { unique: true });
CheckInSchema.index({ userId: 1, createdAt: -1 });

export const CheckIn = model('CheckIn', CheckInSchema);
```

### ‚òê 1.3.3 Design Group schema for shared spaces

```typescript
// src/models/Group.ts
import { Schema, model } from 'mongoose';

const GroupSchema = new Schema({
  name: { type: String, required: true, trim: true },
  description: { type: String, maxlength: 1000 },
  code: { type: String, required: true, unique: true, uppercase: true },
  ownerId: { type: Schema.Types.ObjectId, ref: 'User', required: true },
  members: [{
    userId: { type: Schema.Types.ObjectId, ref: 'User' },
    joinedAt: { type: Date, default: Date.now },
    role: { type: String, enum: ['member', 'moderator'], default: 'member' }
  }],
  settings: {
    isPrivate: { type: Boolean, default: false },
    allowDataSharing: { type: Boolean, default: true },
    maxMembers: { type: Number, default: 50 },
  },
  isActive: { type: Boolean, default: true },
}, { timestamps: true });

GroupSchema.index({ code: 1 });
GroupSchema.index({ ownerId: 1 });
GroupSchema.index({ 'members.userId': 1 });

export const Group = model('Group', GroupSchema);
```

### ‚òê 1.3.5 Add database utilities and helpers

```typescript
// src/utils/database.ts
import { Types } from 'mongoose';

export const isValidObjectId = (id: string): boolean => {
  return Types.ObjectId.isValid(id);
};

export const generateGroupCode = (): string => {
  return Math.random().toString(36).substring(2, 8).toUpperCase();
};

export const getDateRange = (days: number = 30) => {
  const endDate = new Date();
  const startDate = new Date();
  startDate.setDate(startDate.getDate() - days);
  
  return {
    start: new Date(startDate.getFullYear(), startDate.getMonth(), startDate.getDate()),
    end: new Date(endDate.getFullYear(), endDate.getMonth(), endDate.getDate(), 23, 59, 59)
  };
};

export const aggregateUserStats = async (userId: string, days: number = 30) => {
  const { CheckIn } = await import('../models/CheckIn');
  const { start, end } = getDateRange(days);
  
  return CheckIn.aggregate([
    {
      $match: {
        userId: new Types.ObjectId(userId),
        date: { $gte: start, $lte: end }
      }
    },
    {
      $group: {
        _id: null,
        avgMood: { $avg: '$mood' },
        avgEnergy: { $avg: '$energy' },
        avgStress: { $avg: '$stress' },
        totalCheckins: { $sum: 1 }
      }
    }
  ]);
};
```

export interface IUser extends Document {
  email: string;

    notifications: boolean;
  };

  updatedAt: Date;
}

export interface ICheckIn extends Document {
  userId: Types.ObjectId;
  date: Date;
  mood: number;
  energy: number;
  stress: number;
  notes?: string;
  tags: string[];
  isPrivate: boolean;
  createdAt: Date;
  updatedAt: Date;
}

export interface IGroup extends Document {
  name: string;
  description?: string;
  code: string;
  ownerId: Types.ObjectId;
  members: Array<{
    userId: Types.ObjectId;
    joinedAt: Date;
    role: 'member' | 'moderator';
  }>;
  settings: {
    isPrivate: boolean;
    allowDataSharing: boolean;
    maxMembers: number;
  };
  isActive: boolean;
  createdAt: Date;
  updatedAt: Date;
}

export interface UserStats {
  avgMood: number;
  avgEnergy: number;
  avgStress: number;
  totalCheckins: number;
}
